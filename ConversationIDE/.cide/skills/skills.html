import { writeFileSync } from 'fs';
import { tmpdir } from 'os';

export function setupMediaHandlers(ipcMain: IpcMain) {
  ipcMain.handle('media:render-mermaid', async (_event, diagram: string) => {
    // Use mermaid-cli or mmdc to render
    const tmpFile = join(tmpdir(), `diagram-${Date.now()}.mmd`);
    writeFileSync(tmpFile, diagram, 'utf-8');
    return new Promise(resolve => {
      const proc = spawn('npx', ['mmdc', '-i', tmpFile, '-o', `${tmpFile}.png`]);

      proc.on('error', () => resolve({ error: 'mmdc not found' }));
    });
  });

    // Route to local SD / Midjourney API / DALL-E

      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt, steps: 20 }),
    });
    return res.json();
  });

import { IpcMain } from 'electron';

async function brightDataFetch(url: string): Promise<string> {
  const token = process.env.BRIGHT_DATA_TOKEN;
  if (!token) throw new Error('Bright Data token not configured');
  const res = await fetch(`https://api.brightdata.com/request?token=${token}&url=${encodeURIComponent(url)}`);
  return res.text();
}

async function tieredScrape(url: string): Promise<string> {
  const tiers = [
    async () => { const r = await fetch(url); return r.text(); },            // Tier 1: direct
    async () => { const r = await fetch(url, { headers: { 'User-Agent': 'Chrome/120' } }); return r.text(); }, // Tier 2
    async () => brightDataFetch(url),                                                                              // Tier 3: proxy
  ];
  for (const tier of tiers) {
    try { return await tier(); } catch { continue; }
  }
  throw new Error('All scraping tiers exhausted');
}

export function setupScrapingHandlers(ipcMain: IpcMain) {
  ipcMain.handle('scraping:fetch', async (_event, url: string) => {
    try {
      const content = await tieredScrape(url);
      return { ok: true, content };
    } catch (e: any) {
      return { ok: false, error: e.message };
    }
  });
}
